Rome

Dependency free, opt-in UI, customizable date (and time) picker!

      
rome(input);
      
    
      
rome(input, { time: false });
      
    
      
rome(input, { date: false });
      
    
      
var picker = rome(input); // rome.find(input) also returns picker

toggle.addEventListener('click', function () {
  if (picker.restore) {
    picker.restore();
  } else {
    picker.destroy();
  }
});
      
    
      
rome(input, { min: '2013-12-30', max: '2014-10-01', time: false,
	disallow: ['2014-01-15']});
      
    
      
rome(input, { min: '2014-04-30 19:45', max: '2014-09-01 08:30',
	disallow: ['2014-05-15 09:00'] });
